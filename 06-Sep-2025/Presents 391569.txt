# Problem: Presents - https://codeforces.com/problemset/problem/136/A

# include <iostream>

using namespace std;

int main(){
    int n; cin >> n;
    int result[n], arr[n];

    for (int i = 0; i < n; ++i)cin >> arr[i];
    for (int i = 0; i < n; ++i)result[arr[i]-1] = i + 1;
    for (int i = 0; i < n; ++i)cout << result[i] << " ";
}
